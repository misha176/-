private bool x_plus(int start_x, int start_y)
        {
            Color col_pixel;
            start_x++;
            col_pixel = bitmap.GetPixel(start_x, start_y);
            if (col_pixel == c_pixel_start | start_x + 2 < bitmap.Width)
            {
                bitmap.SetPixel(start_x, start_y, Color.Red);
                return true;
            }
            else 
            {
                return false;
            }
        }
        private bool x_minus(int start_x, int start_y)
        {
            Color col_pixel;
            start_x--;
            col_pixel = bitmap.GetPixel(start_x, start_y);
            if (col_pixel == c_pixel_start | start_x-2 > 0 )
            {
                bitmap.SetPixel(start_x, start_y, Color.Red);
                return true;
            }
            else
            {
                return false;
            }
        }
        private bool y_plus(int start_x, int start_y)
        {
            Color col_pixel;
            start_y++;
            col_pixel = bitmap.GetPixel(start_x, start_y);
            if (col_pixel == c_pixel_start | start_y + 2 < bitmap.Height)
            {
                bitmap.SetPixel(start_x, start_y, Color.Red);
                return true;
            }
            else
            {
                return false;
            }
        }
        private bool y_minus(int start_x, int start_y)
        {
            Color col_pixel;
            start_y--;
            col_pixel = bitmap.GetPixel(start_x, start_y);
            if (col_pixel != c_pixel_start | start_y - 2 < 0)
            {
                bitmap.SetPixel(start_x, start_y, Color.Red);
                return true;
            }
            else
            {
                return false;
            }
        }
